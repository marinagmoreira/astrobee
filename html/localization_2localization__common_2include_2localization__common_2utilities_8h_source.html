99 template <class LocMsgType>
100 void CombinedNavStateToMsg(const CombinedNavState& combined_nav_state, LocMsgType& loc_msg) {
101  PoseToMsg(combined_nav_state.pose(), loc_msg.pose);
102  msg_conversions::VectorToMsg(combined_nav_state.velocity(), loc_msg.velocity);
103  msg_conversions::VectorToMsg(combined_nav_state.bias().accelerometer(), loc_msg.accel_bias);
104  msg_conversions::VectorToMsg(combined_nav_state.bias().gyroscope(), loc_msg.gyro_bias);
105  TimeToHeader(combined_nav_state.timestamp(), loc_msg.header);
106 }

108 template <class LocMsgType>
109 void CombinedNavStateCovariancesToMsg(const CombinedNavStateCovariances& covariances, LocMsgType& loc_msg) {
110  // Orientation (0-2)
111  msg_conversions::VariancesToCovDiag(covariances.orientation_variances(), &loc_msg.cov_diag[0]);
112 
113  // Gyro Bias (3-5)
114  msg_conversions::VariancesToCovDiag(covariances.gyro_bias_variances(), &loc_msg.cov_diag[3]);
115 
116  // Velocity (6-8)
117  msg_conversions::VariancesToCovDiag(covariances.velocity_variances(), &loc_msg.cov_diag[6]);
118 
119  // Accel Bias (9-11)
120  msg_conversions::VariancesToCovDiag(covariances.accel_bias_variances(), &loc_msg.cov_diag[9]);
121 
122  // Position (12-14)
123  msg_conversions::VariancesToCovDiag(covariances.position_variances(), &loc_msg.cov_diag[12]);
124 }

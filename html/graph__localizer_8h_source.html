195  template <typename FactorType>
196  void DeleteFactors() {
197  int num_removed_factors = 0;
198  for (auto factor_it = graph_.begin(); factor_it != graph_.end();) {
199  if (dynamic_cast<FactorType*>(factor_it->get())) {
200  factor_it = graph_.erase(factor_it);
201  ++num_removed_factors;
202  continue;
203  }
204  ++factor_it;
205  }
206  LogDebug("DeleteFactors: Num removed factors: " << num_removed_factors);
207  }

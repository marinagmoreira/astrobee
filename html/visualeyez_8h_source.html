130  // This algorithm solves the Procrustes problem in that it finds an affine transform
131  // (rotation, translation, scale) that maps the "in" matrix to the "out" matrix
132  // Code from: https://github.com/oleg-alexandrov/projects/blob/master/eigen/Kabsch.cpp
133  // License is that this code is release in the public domain... Thanks, Oleg :)
134  template <typename T>
135  static bool Kabsch(Eigen::Matrix<T, 3, Eigen::Dynamic> in, Eigen::Matrix<T, 3, Eigen::Dynamic> out,
136  Eigen::Transform<T, 3, Eigen::Affine> &A, bool allowScale) {
137  // Default output
138  A.linear() = Eigen::Matrix<T, 3, 3>::Identity(3, 3);
139  A.translation() = Eigen::Matrix<T, 3, 1>::Zero();
140  // A simple check to see that we have a sufficient number of correspondences
141  if (in.cols() < 4) {
142  // ROS_WARN("Visualeyez needs to see at least four LEDs to track");
143  return false;
144  }
145  // A simple check to see that we have a sufficient number of correspondences
146  if (in.cols() != out.cols()) {
147  // ROS_ERROR("Same number of points required in input matrices");
148  return false;
149  }
150  // First find the scale, by finding the ratio of sums of some distances,
151  // then bring the datasets to the same scale.
152  T dist_in = T(0.0), dist_out = T(0.0);
153  for (int col = 0; col < in.cols()-1; col++) {
154  dist_in += (in.col(col+1) - in.col(col)).norm();
155  dist_out += (out.col(col+1) - out.col(col)).norm();
156  }
157  if (dist_in <= T(0.0) || dist_out <= T(0.0))
158  return true;
159  T scale = T(1.0);
160  if (allowScale) {
161  scale = dist_out/dist_in;
162  out /= scale;
163  }
164  // Find the centroids then shift to the origin
165  Eigen::Matrix<T, 3, 1> in_ctr = Eigen::Matrix<T, 3, 1>::Zero();
166  Eigen::Matrix<T, 3, 1> out_ctr = Eigen::Matrix<T, 3, 1>::Zero();
167  for (int col = 0; col < in.cols(); col++) {
168  in_ctr += in.col(col);
169  out_ctr += out.col(col);
170  }
171  in_ctr /= T(in.cols());
172  out_ctr /= T(out.cols());
173  for (int col = 0; col < in.cols(); col++) {
174  in.col(col) -= in_ctr;
175  out.col(col) -= out_ctr;
176  }
177  // SVD
178  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> Cov = in * out.transpose();
179  Eigen::JacobiSVD < Eigen::Matrix < T, Eigen::Dynamic, Eigen::Dynamic > > svd(Cov,
180  Eigen::ComputeThinU | Eigen::ComputeThinV);
181  // Find the rotation
182  T d = (svd.matrixV() * svd.matrixU().transpose()).determinant();
183  if (d > T(0.0))
184  d = T(1.0);
185  else
186  d = T(-1.0);
187  Eigen::Matrix<T, 3, 3> I = Eigen::Matrix<T, 3, 3>::Identity(3, 3);
188  I(2, 2) = d;
189  Eigen::Matrix<T, 3, 3> R = svd.matrixV() * I * svd.matrixU().transpose();
190  // The final transform
191  A.linear() = scale * R;
192  A.translation() = scale*(out_ctr - R*in_ctr);
193  // Success
194  return true;
195  }
